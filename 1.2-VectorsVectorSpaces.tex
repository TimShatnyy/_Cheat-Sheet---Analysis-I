\subsection{Vectors and Vector Spaces}

\tsDef{(inner product)}{
    For $x,y \in \mathbb{R}^n$ the standard inner product is

    \[
        x \cdot y = \langle x,y\rangle = \sum_{i=1}^{n} x_i y_i
    \]
}
\\
\tsDef{(Euclidean norm)}{
    The Euclidean norm is

    \[
        \|x\| = \sqrt{\sum_{i=1}^{n} x_i^2}
    \]
}
\\
\tsDef{(Euclidean distance)}{
    The Euclidean distance is

    \[
        d(x,y)=\sqrt{\sum_{i=1}^{n}(x_i-y_i)^2}
    \]
}
\tsDef{(Triangle inequality)}{
    For all $x,y,z\in\mathbb{R}^n$:

    \[
        \|x-z\|\le \|x-y\|+\|y-z\|
    \]
}
